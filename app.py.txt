import streamlit as st
import pandas as pd
import joblib

# Load model and scaler
model = joblib.load("best_churn_model.pkl")
scaler = joblib.load("best_churn_scaler.pkl")

st.set_page_config(
    page_title="Telecom Churn Predictor",
    page_icon="📡",
    layout="wide"
)

st.title("📡 Telecom Customer Churn Prediction")
st.markdown(
    """
    This demo predicts whether a telecom customer is likely to churn using a trained
    **Random Forest machine learning model**.
    """
)

st.sidebar.header("Customer Input Details")

call_failure = st.sidebar.number_input("Call Failure", 0, 50, 5)
complains = st.sidebar.selectbox("Complains", [0, 1])
subscription_length = st.sidebar.number_input("Subscription Length", 0, 100, 30)
charge_amount = st.sidebar.number_input("Charge Amount", 0, 10, 1)
seconds_of_use = st.sidebar.number_input("Seconds of Use", 0, 10000, 2000)
frequency_of_use = st.sidebar.number_input("Frequency of Use", 0, 200, 40)
frequency_of_sms = st.sidebar.number_input("Frequency of SMS", 0, 500, 10)
distinct_called_numbers = st.sidebar.number_input("Distinct Called Numbers", 0, 100, 20)
age_group = st.sidebar.number_input("Age Group", 1, 5, 2)
tariff_plan = st.sidebar.number_input("Tariff Plan", 1, 2, 1)
status = st.sidebar.number_input("Status", 1, 2, 1)
age = st.sidebar.number_input("Age", 10, 100, 30)
customer_value = st.sidebar.number_input("Customer Value", 0.0, 5000.0, 150.0)

input_data = pd.DataFrame([{
    "Call  Failure": call_failure,
    "Complains": complains,
    "Subscription  Length": subscription_length,
    "Charge  Amount": charge_amount,
    "Seconds of Use": seconds_of_use,
    "Frequency of use": frequency_of_use,
    "Frequency of SMS": frequency_of_sms,
    "Distinct Called Numbers": distinct_called_numbers,
    "Age Group": age_group,
    "Tariff Plan": tariff_plan,
    "Status": status,
    "Age": age,
    "Customer Value": customer_value
}])

left, right = st.columns([1.2, 1])

with left:
    st.subheader("Customer Details Entered")
    st.dataframe(input_data, use_container_width=True)

    if st.button("Predict Customer Churn", type="primary"):
        scaled_data = scaler.transform(input_data)
        prediction = model.predict(scaled_data)[0]
        probability = model.predict_proba(scaled_data)[0]

        churn_probability = probability[1]
        non_churn_probability = probability[0]

        st.subheader("Prediction Result")

        col1, col2 = st.columns(2)

        with col1:
            st.metric("Probability of Non-Churn", f"{non_churn_probability:.2%}")

        with col2:
            st.metric("Probability of Churn", f"{churn_probability:.2%}")

        if prediction == 1:
            st.error("⚠️ Final Prediction: Customer is likely to churn.")
            st.markdown(
                """
                **Suggested action:** This customer may require retention attention, such as
                service follow-up, complaint resolution, or a loyalty offer.
                """
            )
        else:
            st.success("✅ Final Prediction: Customer is not likely to churn.")
            st.markdown(
                """
                **Suggested action:** This customer appears stable, but continued service quality
                and engagement should be maintained.
                """
            )

with right:
    st.subheader("Project Summary")

    st.markdown(
        """
        **Best Model:** Random Forest  
        **Dataset Used for Demo:** Iranian Churn Dataset  
        **Task Type:** Binary Classification  
        **Target Variable:** Churn  
        """
    )

    st.info(
        "The model predicts whether a customer belongs to the churn or non-churn class."
    )

st.divider()

st.header("Project Visualizations")

chart_col1, chart_col2 = st.columns(2)

with chart_col1:
    st.subheader("Iranian Dataset: Churn Distribution")
    st.image("iran_churn_distribution.png", use_container_width=True)

with chart_col2:
    st.subheader("Bulgarian Dataset: Churn Distribution")
    st.image("bulgarian_churn_distribution.png", use_container_width=True)

chart_col3, chart_col4 = st.columns(2)

with chart_col3:
    st.subheader("Customer Value by Churn")
    st.image("iran_customer_value_by_churn.png", use_container_width=True)

with chart_col4:
    st.subheader("Model Comparison by F1 Score")
    st.image("model_comparison_f1.png", use_container_width=True)

st.divider()

st.header("Model Findings")

st.markdown(
    """
    - The **Random Forest model on the Iranian dataset** performed best overall.
    - The Iranian dataset produced stronger churn prediction results.
    - The Bulgarian dataset showed stronger class imbalance challenges.
    - F1-score, precision, and recall were more useful than accuracy alone.
    """
)

st.caption("Developed by Alungat Patience Otim")
