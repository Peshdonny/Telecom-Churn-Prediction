import streamlit as st
import pandas as pd
import joblib

# Load files
model = joblib.load("best_churn_model.pkl")
scaler = joblib.load("best_churn_scaler.pkl")

st.set_page_config(page_title="Telecom Churn Predictor", page_icon="📡")

st.title("📡 Telecom Customer Churn Prediction")
st.write("Enter customer details to predict whether they are likely to churn.")

# Inputs
call_failure = st.number_input("Call Failure", 0, 50, 5)
complains = st.selectbox("Complains", [0, 1])
subscription_length = st.number_input("Subscription Length", 0, 100, 30)
charge_amount = st.number_input("Charge Amount", 0, 10, 1)
seconds_of_use = st.number_input("Seconds of Use", 0, 10000, 2000)
frequency_of_use = st.number_input("Frequency of Use", 0, 200, 40)
frequency_of_sms = st.number_input("Frequency of SMS", 0, 500, 10)
distinct_called_numbers = st.number_input("Distinct Called Numbers", 0, 100, 20)
age_group = st.number_input("Age Group", 1, 5, 2)
tariff_plan = st.number_input("Tariff Plan", 1, 2, 1)
status = st.number_input("Status", 1, 2, 1)
age = st.number_input("Age", 10, 100, 30)
customer_value = st.number_input("Customer Value", 0.0, 5000.0, 150.0)

if st.button("Predict"):

    data = pd.DataFrame([{
        "Call  Failure": call_failure,
        "Complains": complains,
        "Subscription  Length": subscription_length,
        "Charge  Amount": charge_amount,
        "Seconds of Use": seconds_of_use,
        "Frequency of use": frequency_of_use,
        "Frequency of SMS": frequency_of_sms,
        "Distinct Called Numbers": distinct_called_numbers,
        "Age Group": age_group,
        "Tariff Plan": tariff_plan,
        "Status": status,
        "Age": age,
        "Customer Value": customer_value
    }])

    scaled = scaler.transform(data)

    pred = model.predict(scaled)[0]
    prob = model.predict_proba(scaled)[0]

    if pred == 1:
        st.error("Customer is likely to churn.")
    else:
        st.success("Customer is not likely to churn.")

    st.write(f"Probability of Churn: {prob[1]:.2f}")
